// ТИТРЫ ДЛЯ HALF-LIFE
// Команда позиции $position x y 
// x и y настраиваются от 0 до 1 чтобы не зависеть от разрешения экрана
// -1 означает центр в каждом измерении
// Команда эффектов $effect <номер эффекта>
// Эффект 0 это появление/затухание
// Эффект 1 это мерцающий текст
// Эффект 2 это выписывание (тренировочная комната)
// Цвет текста (красный зелёный синий - r g b) команда $color
// Время появления, время затухания и время удеживания текста
// $fadein (время появления текста - на каждую букву в эффекте 2)
// $fadeout (время затухания текста)
// $holdtime (оставаться на экране на заданную длину времени)


//--------------------- ПЕРЕКЛЮЧАТЕЛИ ОРУЖИЯ
$position 0.5 0.8
$effect 1
$color 128 128 128
$color2 0 0 0
$fadein 0.0
$fadeout 0.0
$holdtime 1.5

Switch_To_BurstFire
{
Переключено в режим стрельбы очередями
}

Switch_To_SemiAuto
{
Переключено в полуавтоматический режим
}

Switch_To_FullAuto
{
Переключено в автоматический режим
}

//-------------------------------------------Сущности провала миссий
$position -1 0.66
$effect 0
$color 128 128 128
$color2 0 0 0
$fadein 0.5
$fadeout 0.5
$holdtime 6

GENERIC_FAIL
{

}

// Название игры
// Это заполнитель для элементов управления эффектами для заголовка "Half-Life".
$position -1 -1
// Только эффект 0 поддерживается названием игры
$effect 0
// Это конечный цвет
$color 180 180 180
// Не используемый color2
$color2 0 0 0
// Это время которое нужно для эффектов появления/затухания названия игры
$fadein 1.0
// Насколько долго держать сообщение на экране (после того как оно полностью проявится)
$holdtime 3.0
// Сколько времени нужно чтобы сообщение исчезло после удержания
$fadeout 1.5
